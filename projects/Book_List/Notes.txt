Remove Books from Local Storage:
removeBook(isbn) {
    const books = JSON.parse(localStorage.getItem('books'));
    books.forEach((book, i) => {
        if(book.isbn === isbn) books.splice(i, 1);
    });
    localStorage.setItem('books', JSON.stringify(books));
}
